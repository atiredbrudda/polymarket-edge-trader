       def get_positions_for_slug(
           session: Session, slug: str, trader_address: str | None = None
       ) -> list[Position]:
           # Try game first, then tournament, then team
           # Check if slug matches a known game
           query = select(Position).join(
               MarketEntity, Position.market_id == MarketEntity.condition_id
           )

           # Check which entity level this slug matches
           game_match = session.execute(
               select(MarketEntity.game).where(MarketEntity.game == slug).limit(1)
           ).scalar_one_or_none()

           if game_match:
               query = query.where(MarketEntity.game == slug)
           else:
               tournament_match = session.execute(
                   select(MarketEntity.tournament).where(MarketEntity.tournament == slug).limit(1)
               ).scalar_one_or_none()
               if tournament_match:
                   query = query.where(MarketEntity.tournament == slug)
               else:
                   # Assume team — check both team_a and team_b
                   query = query.where(
                       (MarketEntity.team_a == slug) | (MarketEntity.team_b == slug)
                   )

           if trader_address is not None:
               query = query.where(Position.trader_address == trader_address)

           result = session.execute(query)
           return list(result.scalars().all())
       ```

       def _get_positions_for_depth(
           session: Session,
           slug: str,
           taxonomy_depth: int,
           trader_address: str | None = None,
       ) -> tuple[list[Any], Decimal]:
           positions = get_positions_for_slug(session, slug, trader_address=trader_address)

           if not positions:
               return [], Decimal("0")

           if taxonomy_depth <= 1:
               return positions, Decimal("0")

           # For depth 2 (tournament): parent is the game
           # For depth 3 (team): parent is the tournament
           # Look up the parent entity from MarketEntity
           if taxonomy_depth == 2:
               # slug is a tournament name; find its game
               parent_name = session.execute(
                   select(MarketEntity.game)
                   .where(MarketEntity.tournament == slug)
                   .where(MarketEntity.game.isnot(None))
                   .limit(1)
               ).scalar_one_or_none()
           else:
               # slug is a team name; find its tournament
               parent_name = session.execute(
                   select(MarketEntity.tournament)
                   .where((MarketEntity.team_a == slug) | (MarketEntity.team_b == slug))
                   .where(MarketEntity.tournament.isnot(None))
                   .limit(1)
               ).scalar_one_or_none()

           if parent_name:
               parent_positions = get_positions_for_slug(
                   session, parent_name, trader_address=trader_address
               )
               parent_volume = sum(
                   (_compute_position_volume(p) for p in parent_positions), Decimal("0")
               )
           else:
               parent_volume = Decimal("0")

           return positions, parent_volume
       ```
