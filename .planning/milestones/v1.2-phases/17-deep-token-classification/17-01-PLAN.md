  <behavior>
    _extract_classification(tags: list[dict]) -> tuple[str | None, int | None]:
      - tags=[] → (None, None)
      - tags=[{slug:"esports"}] only → (None, None)  # no sub-classification available
      - tags=[{slug:"esports"}, {slug:"cs2"}] → ("esports/cs2", 1)
      - tags=[{slug:"esports"}, {slug:"cs2"}, {slug:"iem-katowice-2024"}] → ("esports/cs2/iem-katowice-2024", 2)
      - tags=[{slug:"esports"}, {slug:"cs2"}, {slug:"iem-katowice-2024"}, {slug:"navi"}] → ("esports/cs2/iem-katowice-2024/navi", 3)
      - Tags with missing/empty slug fields are skipped
      - depth is capped at 3 (team level)

    classify_tokens_from_gamma_events(session: Session) -> dict[str, int]:
      - Returns {"classified": N, "skipped_no_tags": M, "skipped_no_tokens": K, "skipped_shallow": L}
      - For each GammaEvent row: parse tags JSON, extract classification, parse clob_token_ids JSON
      - For each token_id in clob_token_ids: UPDATE token_catalog SET node_path=:node_path, depth=:depth WHERE token_id=:token_id AND niche_slug='esports' AND (depth IS NULL OR depth < :depth)
      - Bulk-executes via session.execute(text(...), rows_list) — single batch call, not per-row
      - Commits after all updates
      - Events with NULL/invalid tags JSON → skipped_no_tags
      - Events with only root esports tag (no sub-classification) → skipped_shallow
      - Events with NULL/invalid clob_token_ids JSON → skipped_no_tokens
      - The AND (depth IS NULL OR depth < :depth) guard ensures idempotency and prevents overwriting deeper classifications with shallower ones
  </behavior>

    GREEN phase — implement src/gamma/classification.py:

    Module docstring: "Token classification from Gamma event tags."
    Imports: json, loguru.logger, sqlalchemy.orm.Session, sqlalchemy.text, src.db.models.GammaEvent

    _extract_classification(tags: list[dict]) -> tuple[str | None, int | None]:
      - tag_slugs = [t.get("slug", "").strip() for t in tags if t.get("slug", "").strip()]
      - non_esports = [s for s in tag_slugs if s != "esports"]
      - if not non_esports: return None, None
      - depth = min(len(non_esports), 3)
      - path_parts = ["esports"] + non_esports[:depth]
      - node_path = "/".join(path_parts)
      - return node_path, depth

    classify_tokens_from_gamma_events(session: Session) -> dict[str, int]:
      - Initialize counters: classified=0, skipped_no_tags=0, skipped_no_tokens=0, skipped_shallow=0
      - events = session.query(GammaEvent).all()
      - logger.info(f"Processing {len(events)} gamma events for token classification")
      - update_rows = []
      - For each event:
          - If not event.tags: skipped_no_tags += 1; continue
          - Try json.loads(event.tags) → tags list; except → skipped_no_tags += 1; continue
          - node_path, depth = _extract_classification(tags)
          - If node_path is None: skipped_shallow += 1; continue
          - If not event.clob_token_ids: skipped_no_tokens += 1; continue
          - Try json.loads(event.clob_token_ids) → token_ids list; except → skipped_no_tokens += 1; continue
          - For each token_id in token_ids if token_id: append {"token_id": token_id, "node_path": node_path, "depth": depth}
      - If update_rows:
          session.execute(
              text("""
                  UPDATE token_catalog
                  SET node_path = :node_path,
                      depth = :depth
                  WHERE token_id = :token_id
                    AND niche_slug = 'esports'
                    AND (depth IS NULL OR depth < :depth)
              """),
              update_rows,
          )
          classified = len(update_rows)
      - session.commit()
      - logger.info(f"Classification complete: {classified} token updates, ...")
      - return {"classified": classified, "skipped_no_tags": skipped_no_tags, "skipped_no_tokens": skipped_no_tokens, "skipped_shallow": skipped_shallow}

    Add classify-tokens command to src/cli/commands.py:

      @cli.command("classify-tokens")
      @click.option("--verbose", "-v", is_flag=True, help="Enable debug logging")
      def classify_tokens(verbose):
          """Classify tokens in token_catalog using Gamma event tags.

          Reads gamma_events table (populated by ingest-events) and sets
          token_catalog.node_path and token_catalog.depth for each token
          linked to a Gamma event with sub-classification tags (game, tournament, team).

          node_path format: slash-separated lowercase slugs (e.g., 'esports/cs2' or
          'esports/cs2/iem-katowice-2024'). depth: 1=game, 2=tournament, 3=team.

          Only updates tokens where the new depth is greater than existing depth,
          preserving deeper classifications if run multiple times (idempotent).

          Safe to re-run. Run after 'polymarket ingest-events'.

          Examples:
              polymarket classify-tokens
              polymarket classify-tokens --verbose
          """
          logger.info("CLASSIFY-TOKENS command started")

          if verbose:
              logger.remove()
              logger.add(sys.stderr, level="DEBUG")

          console = Console()

          try:
              settings = get_settings()
              session_factory, _, _, _, _ = _get_dependencies(settings)

              console.print("[bold]Classifying tokens from Gamma event tags...[/bold]")

              with get_session(session_factory) as session:
                  result = classify_tokens_from_gamma_events(session)

              classified = result["classified"]
              skipped_shallow = result["skipped_shallow"]
              skipped_no_tags = result["skipped_no_tags"]
              skipped_no_tokens = result["skipped_no_tokens"]

              console.print(f"[green]Done.[/green] {classified} token updates applied.")
              console.print(f"  Events skipped (no sub-classification tags): [yellow]{skipped_shallow}[/yellow]")
              console.print(f"  Events skipped (no tags):      [yellow]{skipped_no_tags}[/yellow]")
              console.print(f"  Events skipped (no token IDs): [yellow]{skipped_no_tokens}[/yellow]")
              logger.info(
                  f"CLASSIFY-TOKENS completed: {classified} token updates, "
                  f"{skipped_shallow} shallow, {skipped_no_tags} no_tags, "
                  f"{skipped_no_tokens} no_tokens"
              )

          except Exception as e:
              logger.error(f"classify-tokens failed: {e}")
              console.print(f"[red]Error: {e}[/red]")
              raise SystemExit(1)

    REFACTOR phase — if needed: clean up any redundant try/except blocks, ensure logger messages are consistent.
