    def test_idempotent_re_run(self):
        """Idempotency: running resolution twice gives same market.outcome both times."""
        from sqlalchemy import create_engine, text
        from sqlalchemy.orm import sessionmaker
        from src.db.models import Base, GammaEvent, Market

        engine = create_engine("sqlite:///:memory:")
        Base.metadata.create_all(engine)
        Session = sessionmaker(bind=engine)
        session = Session()

        try:
            # Insert a Market row with tok_a as winning token
            market = Market(
                condition_id="cond_1",
                question_id="q_1",
                question="Will team A win?",
                tokens=json.dumps([
                    {"token_id": "tok_a", "outcome": "Yes"},
                    {"token_id": "tok_b", "outcome": "No"},
                ]),
                outcome=None,
            )
            session.add(market)

            # Insert GammaEvent where tok_a wins (price=0.99)
            event = GammaEvent(
                event_id="evt_1",
                clob_token_ids=json.dumps(["tok_a", "tok_b"]),
                outcome_prices=json.dumps(["0.99", "0.01"]),
            )
            session.add(event)
            session.commit()

            # First run: outcome should be set to YES
            result1 = resolve_market_outcomes(session)
            session.expire_all()  # force re-read from DB
            refreshed = session.query(Market).filter_by(condition_id="cond_1").first()
            assert refreshed.outcome == "YES", f"After first run: expected YES, got {refreshed.outcome}"

            # Second run: outcome must NOT change to NO (idempotency)
            result2 = resolve_market_outcomes(session)
            session.expire_all()
            refreshed2 = session.query(Market).filter_by(condition_id="cond_1").first()
            assert refreshed2.outcome == "YES", f"After second run: expected YES, got {refreshed2.outcome}"

            # Both runs should report same resolution count
            assert result1["resolved"] == result2["resolved"]
        finally:
            session.close()
