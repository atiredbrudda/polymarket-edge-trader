```python
from datetime import datetime, timezone

# Polygon mainnet block → Unix timestamp anchors (fetched 2026-02-21)
# Used for piecewise linear interpolation since JBecker parquet has no timestamps
_POLYGON_BLOCK_ANCHORS: list[tuple[int, int]] = [
    (40_000_000, 1678035645),  # 2023-03-05 17:00:45 UTC
    (50_000_000, 1700108689),  # 2023-11-16 04:24:49 UTC
    (60_000_000, 1722368382),  # 2024-07-30 19:39:42 UTC
    (70_000_000, 1744013119),  # 2025-04-07 08:05:19 UTC
    (80_000_000, 1765105043),  # 2025-12-07 10:57:23 UTC
]


def block_number_to_timestamp(block_number: int) -> datetime:
    """Convert a Polygon block number to an approximate UTC datetime.

    Uses piecewise linear interpolation between hardcoded anchor points.
    Accurate to within a few minutes for blocks in the JBecker range (40M–82M).
    Extrapolates linearly beyond the anchor range.

    Args:
        block_number: Polygon mainnet block number

    Returns:
        Approximate UTC datetime (timezone-naive) for when the block was mined
    """
    anchors = _POLYGON_BLOCK_ANCHORS

    # Find surrounding anchor segment
    for i in range(len(anchors) - 1):
        b0, t0 = anchors[i]
        b1, t1 = anchors[i + 1]
        if b0 <= block_number <= b1:
            frac = (block_number - b0) / (b1 - b0)
            unix_ts = t0 + frac * (t1 - t0)
            return datetime.fromtimestamp(unix_ts, tz=timezone.utc).replace(tzinfo=None)

    # Extrapolate from nearest segment
    if block_number < anchors[0][0]:
        b0, t0 = anchors[0]
        b1, t1 = anchors[1]
    else:
        b0, t0 = anchors[-2]
        b1, t1 = anchors[-1]

    frac = (block_number - b0) / (b1 - b0)
    unix_ts = t0 + frac * (t1 - t0)
    return datetime.fromtimestamp(unix_ts, tz=timezone.utc).replace(tzinfo=None)
```

```python
@cli.command("reset-backfill")
@click.option("--confirm", is_flag=True, help="Skip confirmation prompt")
@click.option("--verbose", "-v", is_flag=True, help="Enable debug logging")
def reset_backfill(confirm, verbose):
    """Clear JBecker trades and reset backfill status for re-ingestion.

    Deletes all JBecker-sourced trades from the database and marks affected
    traders as pending backfill (backfill_complete=False). Run this before
    re-running 'polymarket backfill' after a timestamp fix.

    \b
    Examples:
        polymarket reset-backfill
        polymarket reset-backfill --confirm
    """
    logger.info("RESET-BACKFILL command started")

    if verbose:
        logger.remove()
        logger.add(sys.stderr, level="DEBUG")

    console = Console()

    session_factory, _, _, _, _ = _get_dependencies()

    with get_session(session_factory) as session:
        jbecker_count = (
            session.query(Trade).filter(Trade.trade_id.like("jbecker_%")).count()
        )

        if jbecker_count == 0:
            console.print("[green]No JBecker trades found — nothing to reset.[/green]")
            return

        console.print(
            f"[yellow]This will delete {jbecker_count:,} JBecker trades and reset backfill status.[/yellow]"
        )

        if not confirm and not click.confirm("Continue?"):
            console.print("[dim]Aborted.[/dim]")
            return

        # Find affected trader addresses before deleting trades
        affected_rows = (
            session.query(Trade.trader_address)
            .filter(Trade.trade_id.like("jbecker_%"))
            .distinct()
            .all()
        )
        affected_addresses = [row[0] for row in affected_rows]

        # Delete JBecker trades
        deleted = (
            session.query(Trade)
            .filter(Trade.trade_id.like("jbecker_%"))
            .delete(synchronize_session=False)
        )

        # Reset backfill_complete for affected traders
        if affected_addresses:
            session.query(Trader).filter(
                Trader.address.in_(affected_addresses)
            ).update({"backfill_complete": False}, synchronize_session=False)

        session.commit()

    console.print(f"\n[bold green]Reset complete[/bold green]")
    console.print(f"  Trades deleted:   [red]{deleted:,}[/red]")
    console.print(f"  Traders reset:    [yellow]{len(affected_addresses)}[/yellow]")
    console.print(
        "\n[dim]Run 'polymarket backfill' to re-ingest with corrected timestamps.[/dim]"
    )
    logger.info(
        f"RESET-BACKFILL completed: {deleted} trades deleted, {len(affected_addresses)} traders reset"
    )
```
