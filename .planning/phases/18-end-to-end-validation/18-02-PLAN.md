    ```python
    def backfill_market_classifications(session: Session) -> dict[str, int]:
        """Update MarketClassification.taxonomy_node_id for rows where
        token_catalog has a node_path but MarketClassification.taxonomy_node_id is NULL.

        This fixes the forward-only population problem: classify-tokens updates
        token_catalog but cannot update MarketClassification rows that already
        existed with NULL taxonomy_node_id before classification ran.

        Caller must commit after this returns.
        """
        from src.db.models import MarketClassification, TokenCatalog, TaxonomyNode

        updated = 0
        skipped_no_node = 0

        rows = (
            session.query(MarketClassification, TokenCatalog)
            .join(TokenCatalog, MarketClassification.market_id == TokenCatalog.condition_id)
            .filter(MarketClassification.taxonomy_node_id.is_(None))
            .filter(TokenCatalog.node_path.is_not(None))
            .all()
        )

        logger.info(f"Backfill: found {len(rows)} MarketClassification rows needing update")

        for mc, tc in rows:
            # node_path format: "esports/cs2" — TaxonomyNode.slug format: "esports.cs2"
            slug = tc.node_path.replace("/", ".")
            node = session.query(TaxonomyNode).filter_by(slug=slug).first()
            if node:
                mc.taxonomy_node_id = node.id
                mc.node_path = tc.node_path
                updated += 1
            else:
                skipped_no_node += 1

        logger.info(
            f"Classification backfill: {updated} updated, {skipped_no_node} no matching node"
        )
        return {"updated": updated, "skipped_no_node": skipped_no_node}
    ```

    @cli.command("backfill-classifications")
    @click.option("--verbose", "-v", is_flag=True, help="Enable debug logging")
    def backfill_classifications(verbose):
        """Update MarketClassification.taxonomy_node_id for rows with NULL node IDs.

        Fixes the forward-only population problem: tokens classified by
        classify-tokens that had NULL node_path at JBecker backfill time will
        have their MarketClassification.taxonomy_node_id updated here.

        Safe to re-run — skips rows that already have taxonomy_node_id set.

        Examples:
            polymarket backfill-classifications
        """
        logger.info("BACKFILL-CLASSIFICATIONS command started")

        if verbose:
            logger.remove()
            logger.add(sys.stderr, level="DEBUG")

        console = Console()

        try:
            settings = get_settings()
            session_factory, _, _, _, _ = _get_dependencies(settings)

            console.print("[bold]Backfilling MarketClassification taxonomy node IDs...[/bold]")

            with get_session(session_factory) as session:
                counts = backfill_market_classifications(session)
                session.commit()

            updated = counts["updated"]
            skipped_no_node = counts["skipped_no_node"]

            console.print(f"[green]Done.[/green] {updated} classifications updated.")
            console.print(
                f"  Skipped (no matching taxonomy node): [yellow]{skipped_no_node}[/yellow]"
            )
            logger.info(
                f"BACKFILL-CLASSIFICATIONS completed: {updated} updated, {skipped_no_node} skipped"
            )

        except Exception as e:
            logger.error(f"backfill-classifications failed: {e}")
            console.print(f"[red]Error: {e}[/red]")
            raise SystemExit(1)
